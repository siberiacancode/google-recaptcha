import { Steps, Step } from 'fumadocs-ui/components/steps';

<Steps>
  <Step title="Installation">
    Install the required package using your preferred package manager:

    ```package-install
    npm install @google-recaptcha/react
    ```

  </Step>

  <Step title="Provider Setup">
    Wrap your application with the `GoogleReCaptchaProvider` in your root component:

    ```tsx
    // src/main.tsx
    import React from 'react';
    import ReactDOM from 'react-dom/client';
    import { GoogleReCaptchaProvider } from '@google-recaptcha/react'; // [!code highlight]

    import App from './App';

    ReactDOM.createRoot(document.getElementById('root')!).render(
      <React.StrictMode>
        <GoogleReCaptchaProvider
          type="v3"
          siteKey="your_site_key"
        >
          <App />
        </GoogleReCaptchaProvider>
      </React.StrictMode>
    );
    ```

  </Step>

  <Step title="Client Component Usage">
    Create a component to use reCAPTCHA:

    ```tsx
    import type { FormEvent } from 'react';
    import { useGoogleReCaptcha } from '@google-recaptcha/react';

    export const Form = () => {
      const googleReCaptcha = useGoogleReCaptcha();

      const onSubmit = async (event: FormEvent) => {
        event.preventDefault();

        const token = await googleReCaptcha.executeV3();
        ...
      };

      return (
        <form onSubmit={onSubmit}>
          ...
          <button type="submit">Submit</button>
        </form>
      );
    }
    ```

  </Step>
</Steps>
